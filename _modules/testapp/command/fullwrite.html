    def run(self, value: str | int) -> None:
        """
        SSD의 모든 LBA에 값을 씁니다.

        Args:
            value (str): 쓸 값 (16진수 문자열)
        """
        for i in SSD_LBA_RANGE:
            self.driver.write(i, value)
